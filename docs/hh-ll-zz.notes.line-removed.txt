indicator("HH-LL ZZ", shorttitle='H²L²Z²', max_lines_count=500, max_labels_count=500, max_bars_back=1000, overlay=true)

// https://www.tradingview.com/pine-script-reference/v5/
// https://www.tradingview.com/pine-script-reference/v5/
// https://www.tradingview.com/pine-script-reference/v5/

// options and default values
cl      = input.string(     'close'    , 'source for level breach', options=['H/L', 'close']                                        )
cnt     = input.int   (        3       , 'x breaches'             , minval=1, maxval=10                                             )
style   = input.string(line.style_solid, 'line style'             , options=[line.style_solid, line.style_dotted, line.style_dashed])
showLev = input.bool  (      false     , 'show levels breaches'                                                                     )
showS_R = input.bool  (      false     , 'show Support/Resistance'                                                                  )
ds      = input.bool  (      false     , 'remove repaint warning'                                                                   )
lab     = input.bool  (      false     , 'labels'                                                                                   )
lin     = input.bool  (      true      , 'lines'                                                                                    )

// vars
var label[] labels  = array.new<label>()
var float   levelH  = high // current level high
var float   levelL  = low  // current level low
var int     countLL = 0    // count of lower lows
var int     countHH = 0    // count of higher hights
var int     dir     = 0    // current direction, 1=up, -1=down
bool        h       = false 
bool        l       = false 


// if first bar in series, add a new label with x=bar_index, y=(high+low)/2

if barstate.isfirst 
    array.unshift(labels, label.new(bar_index, hl2))


// if the bar close is lower than the current level low then handle the case going lower case

if (cl == 'H/L' ? low : close) < levelL 

    // increase the count of lower lowers

    countLL += 1

    // if the count of lower lows is equal to the user defined option of x-breachs

    if countLL  ==  cnt
        // set low is true

        l       :=  true
        
        // reset the counts of lower lowers and higher highers

        countHH :=  0
        countLL :=  0
        
        // set the current level low and current level high to the low and high of this bar
        
        levelL  :=  low
        levelH  :=  high

        // if direction was up (which it is by default)

        if dir   > -1
        
            // new up a label with x=bar_index, y=(high+low)/2
            // prepend this label to the head of the labels array
        
            array.unshift(labels, label.new(bar_index, low, style=label.style_label_up, color=lab ? #FF0000 : color.new(color.blue, 100)))
            
            // set the current direction to down

            dir := -1
        else
            // the current direction is up

            // label.set_xy(id, x, y) → void
            
            // get the label at the head of the labels array, mutate it by setting x to this bars index and y to this bars low

            label.set_xy(array.get(labels, 0), bar_index, low)


        // if there are more than 2 labels in the labels array

        if array.size(labels) > 2

            // instantiate a variable, hi, and set it to this bars low
            
            hi = low

            // instantiate a variable, bx (bar_x) and set it to 0

            bx = 0

            // iterate over the bars in the barset from this bars index, to the bar index of the label in labels at labels[2] 

            for i =  0 to bar_index - label.get_x(array.get(labels, 2))
            
                // if the high of the bar is high than hi

                if high[i] > hi

                    // set the hi to the high of i

                    hi := high[i]
                    
                    // set the bx (bar x) to the value of this bars index - 1

                    bx := bar_index - i
            
            // get the label at labels[1] and mutate it to set xy x=bx, y=hi

            label.set_xy(array.get(labels, 1), bx, hi)


// handle the case where the high or the lower if greater than the level high
// as with the previous case, but opposit

if (cl == 'H/L' ? high : close) > levelH 
    countHH += 1
    if countHH  ==  cnt
        h       :=  true
        countHH :=  0
        countLL :=  0
        levelL  :=  low
        levelH  :=  high
        if dir   <  1
            array.unshift(labels, label.new(bar_index, high, style=label.style_label_down, color=lab ? #FF0000 : color.new(color.blue, 100)))
            dir :=  1
            //Added as extra prevention repaint yellow line/label
            if barstate.isconfirmed
                if array.size(labels) > 2
                    label.set_color(array.get(labels, 2), lab ? color.yellow : color.new(color.blue, 100))
        else
            label.set_xy(array.get(labels, 0), bar_index, high)
        if array.size(labels) > 2   
            lo = high
            bx = 0  
            for i =  0 to bar_index - label.get_x(array.get(labels, 2)) 
                if low [i] < lo
                    lo := low[i]
                    bx := bar_index - i
            label.set_xy(array.get(labels, 1), bx, lo)
